  created_by uuid REFERENCES profiles(id)

CREATE POLICY "order_payments_update" ON order_payments
  FOR UPDATE USING (
    EXISTS (
      SELECT 1 FROM orders
      WHERE orders.id = order_payments.order_id
        AND orders.restaurant_id = get_user_restaurant_id()
    )
  );

-- 2. New table: order_cancellations
CREATE TABLE IF NOT EXISTS order_cancellations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id uuid REFERENCES orders(id) ON DELETE CASCADE,
  order_item_id uuid REFERENCES order_items(id) ON DELETE CASCADE,
  reason text NOT NULL,
  cancelled_by uuid NOT NULL REFERENCES profiles(id),
  cancelled_at timestamptz DEFAULT now(),
  CONSTRAINT at_least_one_target CHECK (order_id IS NOT NULL OR order_item_id IS NOT NULL)
);

-- RLS on order_cancellations
ALTER TABLE order_cancellations ENABLE ROW LEVEL SECURITY;

CREATE POLICY "order_cancellations_select" ON order_cancellations
  FOR SELECT USING (
    (order_id IS NOT NULL AND EXISTS (
      SELECT 1 FROM orders
      WHERE orders.id = order_cancellations.order_id
        AND orders.restaurant_id = get_user_restaurant_id()
    ))
    OR
    (order_item_id IS NOT NULL AND EXISTS (
      SELECT 1 FROM order_items
      JOIN orders ON orders.id = order_items.order_id
      WHERE order_items.id = order_cancellations.order_item_id
        AND orders.restaurant_id = get_user_restaurant_id()

      SELECT 1 FROM orders

        AND orders.restaurant_id = get_user_restaurant_id()

    OR
    (order_item_id IS NOT NULL AND EXISTS (
      SELECT 1 FROM order_items
      JOIN orders ON orders.id = order_items.order_id
      WHERE order_items.id = order_cancellations.order_item_id
        AND orders.restaurant_id = get_user_restaurant_id()
    ))
  );

-- 3. Alter orders: add order_type, customer fields, paid_amount
ALTER TABLE orders
  ADD COLUMN IF NOT EXISTS order_type text CHECK (order_type IN ('dine_in', 'takeaway', 'delivery')) DEFAULT 'dine_in',
  ADD COLUMN IF NOT EXISTS customer_name text,
  ADD COLUMN IF NOT EXISTS customer_phone text,
  ADD COLUMN IF NOT EXISTS delivery_address text,

// ---------------------------------------------------------------------------

export async function cancelOrderItem(
  itemId: string,
  reason: string
): Promise<void> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  // Get current user
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error("Non authentifié");

  // Verify item belongs to this restaurant
  const { data: item } = await supabase
    .from("order_items")
    .select("id, order_id, unit_price, quantity")
    .eq("id", itemId)
    .single();

  if (!item) throw new Error("Article non trouvé");

  const { data: order } = await supabase
    .from("orders")
    .select("id, restaurant_id, total")
    .eq("id", item.order_id)
    .single();

  if (!order || order.restaurant_id !== restaurantId) {
    throw new Error("Commande non trouvée");
  }

  // 1. Update item status to cancelled
  await supabase
    .from("order_items")
    .update({ status: "cancelled" })
    .eq("id", itemId);

  // 2. Insert cancellation record
  await supabase
    .from("order_cancellations")
    .insert({
      order_item_id: itemId,
      order_id: item.order_id,
      reason,
      cancelled_by: user.id,
    });

  // 3. Recalculate order total (subtract cancelled item)
  const newTotal = (order.total ?? 0) - item.unit_price * item.quantity;
  await supabase
    .from("orders")
    .update({ total: Math.max(0, newTotal), updated_at: new Date().toISOString() })
    .eq("id", order.id);

  // 4. Clean up preparation ticket
  const { data: ticketItems } = await supabase
    .from("order_items")
    .select("id, preparation_ticket_id, status")
    .eq("order_id", item.order_id)
    .not("status", "eq", "cancelled");

  // Find tickets that now have no non-cancelled items
  const { data: allTickets } = await supabase
    .from("preparation_tickets")
    .select("id")
    .eq("order_id", item.order_id);

  for (const ticket of allTickets ?? []) {
    const ticketHasItems = (ticketItems ?? []).some(
      (ti) => ti.preparation_ticket_id === ticket.id
    );
    if (!ticketHasItems) {
      await supabase
        .from("preparation_tickets")
        .update({ status: "served" })
        .eq("id", ticket.id);
    }
  }

  // 5. Re-evaluate order status
  const remainingItems = (ticketItems ?? []).filter((ti) => ti.status !== "cancelled");
  if (remainingItems.length === 0) {
    await supabase
      .from("orders")
      .update({ status: "cancelled", updated_at: new Date().toISOString() })
      .eq("id", order.id);
  }
}

export async function cancelOrder(
  orderId: string,
  reason: string
): Promise<void> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  // Get current user
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error("Non authentifié");

  // Verify permission: owner/admin/manager only
  const { data: profile } = await supabase
    .from("profiles")
    .select("role")
    .eq("id", user.id)
    .single();

  if (!profile || !["owner", "admin", "manager"].includes(profile.role ?? "")) {
    throw new Error("Seuls les managers peuvent annuler une commande entière");
  }

  // Verify order belongs to this restaurant
  const { data: order } = await supabase
    .from("orders")
    .select("id, restaurant_id, status")
    .eq("id", orderId)
    .eq("restaurant_id", restaurantId)
    .single();

  if (!order) throw new Error("Commande non trouvée");
  if (order.status === "paid" || order.status === "cancelled") {
    throw new Error("Impossible d'annuler une commande déjà payée ou annulée");
  }

  // 1. Cancel order
  await supabase
    .from("orders")
    .update({ status: "cancelled", updated_at: new Date().toISOString() })
    .eq("id", orderId);

  // 2. Insert cancellation record
  await supabase
    .from("order_cancellations")
    .insert({
      order_id: orderId,
      reason,
      cancelled_by: user.id,
    });

  // 3. Cancel all items
  await supabase
    .from("order_items")
    .update({ status: "cancelled" })
    .eq("order_id", orderId);

  // 4. Release all tickets from KDS
  await supabase
    .from("preparation_tickets")
    .update({ status: "served" })
    .eq("order_id", orderId)
    .in("status", ["pending", "in_progress", "ready"]);
}
```

export async function createPayment(
  orderId: string,
  data: {
    amount: number;
    method: "cash" | "card" | "transfer" | "other";
    label?: string;
    itemIds?: string[];
  }
): Promise<{ paid_amount: number; is_fully_paid: boolean }> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new Error("Non authentifié");

  // Verify order
  const { data: order } = await supabase
    .from("orders")
    .select("id, restaurant_id, total, paid_amount, status")
    .eq("id", orderId)
    .eq("restaurant_id", restaurantId)
    .single();

  if (!order) throw new Error("Commande non trouvée");

  // Insert payment
  const { data: payment, error: paymentError } = await supabase
    .from("order_payments")
    .insert({
      order_id: orderId,
      amount: data.amount,
      method: data.method,
      label: data.label ?? null,
      created_by: user.id,
    })
    .select()
    .single();

  if (paymentError) throw new Error(`Erreur paiement: ${paymentError.message}`);

  // Update paid_amount on order
  const newPaidAmount = (order.paid_amount ?? 0) + data.amount;
  const isFullyPaid = newPaidAmount >= (order.total ?? 0);

  const orderUpdate: Record<string, unknown> = {
    paid_amount: newPaidAmount,
    updated_at: new Date().toISOString(),
  };

  if (isFullyPaid) {
    orderUpdate.status = "paid";
  }

  await supabase
    .from("orders")
    .update(orderUpdate)
    .eq("id", orderId);

  // If specific items were paid, link them to this payment
  if (data.itemIds && data.itemIds.length > 0) {
    await supabase
      .from("order_items")
      .update({ payment_id: payment.id })
      .in("id", data.itemIds);
  }

  return { paid_amount: newPaidAmount, is_fully_paid: isFullyPaid };
}

export async function getOrderPayments(
  orderId: string
): Promise<{ id: string; amount: number; method: string; label: string | null; paid_at: string }[]> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  // Verify order belongs to restaurant
  const { data: order } = await supabase
    .from("orders")
    .select("id")
    .eq("id", orderId)
    .eq("restaurant_id", restaurantId)
    .single();

  if (!order) throw new Error("Commande non trouvée");

  const { data, error } = await supabase
    .from("order_payments")
    .select("id, amount, method, label, paid_at")
    .eq("order_id", orderId)
    .order("paid_at", { ascending: true });

  if (error) throw new Error(`Erreur chargement paiements: ${error.message}`);
  return (data ?? []).map((p) => ({
    ...p,
    amount: Number(p.amount),
    paid_at: p.paid_at ?? new Date().toISOString(),
  }));
}
```

export async function getOrderStats(date: string): Promise<OrderStats> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  const { data, error } = await supabase
    .from("orders")
    .select("status, total, order_type")
    .eq("restaurant_id", restaurantId)
    .gte("created_at", date)
    .lt("created_at", `${date}T23:59:59.999Z`);

  if (error) {
    throw new Error(
      `Erreur lors du chargement des statistiques : ${error.message}`
    );
  }

  const rows = data ?? [];
  const activeStatuses: OrderStatus[] = ["sent", "preparing", "ready"];
  const completedStatuses: OrderStatus[] = ["served", "paid"];

  let revenue = 0;
  let activeOrders = 0;
  let completedOrders = 0;
  const breakdown = {
    dine_in: { count: 0, revenue: 0 },
    takeaway: { count: 0, revenue: 0 },
    delivery: { count: 0, revenue: 0 },
  };

  for (const row of rows) {
    const status = (row.status ?? "draft") as OrderStatus;
    const orderTotal = row.total ?? 0;
    revenue += orderTotal;

    if (activeStatuses.includes(status)) activeOrders++;
    else if (completedStatuses.includes(status)) completedOrders++;

    const orderType = (row.order_type ?? "dine_in") as keyof typeof breakdown;
    if (orderType in breakdown) {
      breakdown[orderType].count++;
      breakdown[orderType].revenue += orderTotal;
    }
  }

  return { totalOrders: rows.length, revenue, activeOrders, completedOrders, breakdown };
}
```

export interface CancellationRecord {
  id: string;

  cancelled_at: string;
  cancelled_by_name: string;

export async function getCancellations(): Promise<CancellationRecord[]> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  const { data, error } = await supabase
    .from("order_cancellations")
    .select("*")
    .order("cancelled_at", { ascending: false })
    .limit(200);

  if (error) throw new Error(`Erreur chargement annulations: ${error.message}`);
  if (!data || data.length === 0) return [];

  // Get related order info
  const orderIds = [...new Set(data.map((c) => c.order_id).filter(Boolean))] as string[];
  const itemIds = [...new Set(data.map((c) => c.order_item_id).filter(Boolean))] as string[];
  const userIds = [...new Set(data.map((c) => c.cancelled_by))];

  const [ordersRes, itemsRes, profilesRes] = await Promise.all([
    orderIds.length > 0
      ? supabase.from("orders").select("id, table_number, restaurant_id").in("id", orderIds)
      : { data: [] },
    itemIds.length > 0
      ? supabase.from("order_items").select("id, product_name").in("id", itemIds)
      : { data: [] },
    userIds.length > 0
      ? supabase.from("profiles").select("id, full_name").in("id", userIds)
      : { data: [] },
  ]);

  const orderMap = new Map((ordersRes.data ?? []).map((o) => [o.id, o]));
  const itemMap = new Map((itemsRes.data ?? []).map((i) => [i.id, i]));
  const profileMap = new Map((profilesRes.data ?? []).map((p) => [p.id, p]));

  return data
    .filter((c) => {
      // Filter by restaurant
      if (c.order_id) {
        const order = orderMap.get(c.order_id);
        return order?.restaurant_id === restaurantId;
      }
      return true;
    })
    .map((c) => {
      const order = c.order_id ? orderMap.get(c.order_id) : null;
      const item = c.order_item_id ? itemMap.get(c.order_item_id) : null;
      const profile = profileMap.get(c.cancelled_by);
      return {
        id: c.id,
        order_id: c.order_id,
        order_item_id: c.order_item_id,
        reason: c.reason,
        cancelled_at: c.cancelled_at ?? new Date().toISOString(),
        cancelled_by_name: profile?.full_name ?? "Inconnu",
        table_number: order?.table_number ?? null,
        item_name: item?.product_name ?? null,
      };
    });
}

      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
  }

  async function handlePayItems() {
    if (selectedTotal <= 0 || submitting) return;
    setSubmitting(true);
    try {
      const result = await onCreatePayment({
        amount: selectedTotal,
        method,
        label: `${selectedItemIds.size} article(s)`,
        itemIds: [...selectedItemIds],
      });
      setPaidAmount(result.paid_amount);
      setSelectedItemIds(new Set());
      if (result.is_fully_paid) {
        toast.success("Commande encaissée");
        onOpenChange(false);
      } else {
        toast.success(`Paiement de ${selectedTotal.toFixed(2)} EUR enregistré`);
      }
    } catch (error) {
      toast.error(error instanceof Error ? error.message : "Erreur paiement");
    } finally {
      setSubmitting(false);
    }
  }

  async function handlePaySplit() {
    if (splitAmount <= 0 || submitting) return;
    setSubmitting(true);
    try {
      const partLabel = `Part ${paidParts + 1}/${splitCount}`;
      const result = await onCreatePayment({
        amount: splitAmount,
        method,
        label: partLabel,
      });
      setPaidAmount(result.paid_amount);
      setPaidParts((p) => p + 1);
      if (result.is_fully_paid) {
        toast.success("Commande encaissée");
        onOpenChange(false);
      } else {
        toast.success(`${partLabel} — ${splitAmount.toFixed(2)} EUR enregistré`);
      }
    } catch (error) {
      toast.error(error instanceof Error ? error.message : "Erreur paiement");
    } finally {
      setSubmitting(false);
    }
  }

  const cashChange = method === "cash" && cashReceived
    ? parseFloat(cashReceived) - (activeTab === "items" ? selectedTotal : splitAmount)
    : null;

  const tabs: { key: TabKey; label: string; icon: React.ReactNode }[] = [
    { key: "items", label: "Par articles", icon: <Layers className="h-4 w-4" /> },
    { key: "equal", label: "Split egal", icon: <Users className="h-4 w-4" /> },
    { key: "mixed", label: "Mixte", icon: <CreditCard className="h-4 w-4" /> },
  ];

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="max-w-lg max-h-[90vh] overflow-y-auto">
        <DialogHeader>
          <DialogTitle>
            Encaisser — {total.toFixed(2)} EUR
            {paidAmount > 0 && (
              <span className="ml-2 text-sm font-normal text-green-600">
                (déjà payé: {paidAmount.toFixed(2)} EUR)
              </span>
            )}
          </DialogTitle>
        </DialogHeader>

        {/* Remaining */}
        <div className="rounded-lg bg-muted p-3 text-center">
          <p className="text-sm text-muted-foreground">Reste à payer</p>
          <p className="text-2xl font-bold">{remaining.toFixed(2)} EUR</p>
        </div>

        {/* Tabs */}
        <div className="flex gap-1 rounded-lg bg-muted/50 p-1">
          {tabs.map((tab) => (
            <button
              key={tab.key}
              type="button"
              className={cn(
                "flex flex-1 items-center justify-center gap-1.5 rounded-md px-3 py-2 text-sm font-medium transition-colors",
                activeTab === tab.key
                  ? "bg-background shadow-sm"
                  : "text-muted-foreground hover:text-foreground"
              )}
              onClick={() => setActiveTab(tab.key)}
            >
              {tab.icon}
              {tab.label}
            </button>
          ))}
        </div>

        {/* Payment method selector */}
        <div className="flex gap-2">
          {(Object.entries(METHOD_LABELS) as [PaymentMethod, string][]).map(([key, label]) => (
            <Button
              key={key}
              variant={method === key ? "default" : "outline"}
              size="sm"
              className="min-h-9 flex-1"
              onClick={() => setMethod(key)}
            >
              {label}
            </Button>
          ))}
        </div>

        {method === "cash" && (
          <div>
            <label className="text-sm font-medium">Montant recu</label>
            <input
              type="number"
              step="0.01"
              min="0"
              className="mt-1 w-full rounded-lg border px-3 py-2 text-sm"
              value={cashReceived}
              onChange={(e) => setCashReceived(e.target.value)}
              placeholder="0.00"
            />
            {cashChange !== null && cashChange >= 0 && (
              <p className="mt-1 text-sm font-semibold text-green-600">
                Rendu monnaie: {cashChange.toFixed(2)} EUR
              </p>
            )}
          </div>
        )}

        {/* Tab content: Par articles */}
        {(activeTab === "items" || activeTab === "mixed") && (
          <div className="space-y-2">
            <p className="text-sm font-medium text-muted-foreground">
              Sélectionnez les articles à encaisser
            </p>
            <div className="space-y-1 max-h-[300px] overflow-y-auto">
              {unpaidItems.map((item) => {
                const isSelected = selectedItemIds.has(item.id);
                return (
                  <button
                    key={item.id}
                    type="button"
                    className={cn(
                      "flex w-full items-center gap-3 rounded-lg border px-3 py-2.5 text-left transition-colors",
                      isSelected
                        ? "border-primary bg-primary/5"
                        : "hover:bg-muted/50"
                    )}
                    onClick={() => toggleItem(item.id)}
                  >
                    <div
                      className={cn(
                        "flex h-[44px] w-[44px] shrink-0 items-center justify-center rounded-lg border-2 transition-colors",
                        isSelected
                          ? "border-primary bg-primary text-white"
                          : "border-muted"
                      )}
                    >
                      {isSelected && <CreditCard className="h-5 w-5" />}
                    </div>
                    <div className="flex-1">
                      <span className="text-sm font-medium">
                        {item.quantity}x {item.product_name}
                      </span>
                    </div>
                    <span className="text-sm font-semibold">
                      {(item.quantity * item.unit_price).toFixed(2)} EUR
                    </span>
                  </button>
                );
              })}
            </div>
            {selectedTotal > 0 && (
              <div className="flex items-center justify-between rounded-lg bg-primary/5 p-3">
                <span className="text-sm font-medium">Sous-total sélection</span>
                <span className="text-lg font-bold">{selectedTotal.toFixed(2)} EUR</span>
              </div>
            )}
            <Button
              className="min-h-11 w-full"
              disabled={selectedTotal <= 0 || submitting}
              onClick={handlePayItems}
            >
              {submitting ? "Paiement..." : `Valider ce paiement — ${selectedTotal.toFixed(2)} EUR`}
            </Button>
          </div>
        )}

        {/* Tab content: Split egal */}
        {(activeTab === "equal" || (activeTab === "mixed" && paidAmount > initialPaidAmount)) && (
          <div className="space-y-3">
            {activeTab === "mixed" && (
              <div className="rounded-lg bg-amber-50 border border-amber-200 p-2 text-sm text-amber-800">
                Articles individuels encaisses. Split egal sur le solde restant.
              </div>
            )}
            <div className="flex items-center justify-center gap-4">
              <Button
                variant="outline"
                size="icon"
                className="min-h-11 min-w-11"
                onClick={() => setSplitCount(Math.max(2, splitCount - 1))}
              >
                -
              </Button>
              <div className="text-center">
                <p className="text-3xl font-bold">{splitCount}</p>
                <p className="text-xs text-muted-foreground">convives</p>
              </div>
              <Button
                variant="outline"
                size="icon"
                className="min-h-11 min-w-11"
                onClick={() => setSplitCount(Math.min(12, splitCount + 1))}
              >
                +
              </Button>
            </div>

            <div className="rounded-lg bg-muted p-3 text-center">
              <p className="text-sm text-muted-foreground">Montant par part</p>
              <p className="text-2xl font-bold">{splitAmount.toFixed(2)} EUR</p>
            </div>

            <div className="text-center text-sm font-medium">
              {paidParts}/{splitCount} parts payées
            </div>

            <Button
              className="min-h-11 w-full"
              disabled={paidParts >= splitCount || submitting}
              onClick={handlePaySplit}
            >
              {submitting ? "Paiement..." : `Payer 1 part — ${splitAmount.toFixed(2)} EUR`}
            </Button>
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}
```
