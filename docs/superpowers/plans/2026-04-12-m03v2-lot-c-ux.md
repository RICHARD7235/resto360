-- M03 V2 Lot C: restaurant_tables + seed LCQF

CREATE TABLE IF NOT EXISTS restaurant_tables (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  restaurant_id uuid NOT NULL REFERENCES restaurants(id) ON DELETE CASCADE,
  name text NOT NULL,
  zone text,
  capacity int DEFAULT 4,
  shape text CHECK (shape IN ('square', 'round', 'rectangle')) DEFAULT 'square',
  width int DEFAULT 1,
  height int DEFAULT 1,
  pos_x float DEFAULT 0,
  pos_y float DEFAULT 0,
  is_active boolean DEFAULT true,
  sort_order int DEFAULT 0,
  created_at timestamptz DEFAULT now()
);

-- RLS
ALTER TABLE restaurant_tables ENABLE ROW LEVEL SECURITY;

CREATE POLICY "restaurant_tables_select" ON restaurant_tables
  FOR SELECT USING (restaurant_id = get_user_restaurant_id());

CREATE POLICY "restaurant_tables_insert" ON restaurant_tables
  FOR INSERT WITH CHECK (restaurant_id = get_user_restaurant_id());

CREATE POLICY "restaurant_tables_update" ON restaurant_tables
  FOR UPDATE USING (restaurant_id = get_user_restaurant_id());

CREATE POLICY "restaurant_tables_delete" ON restaurant_tables
  FOR DELETE USING (restaurant_id = get_user_restaurant_id());

-- Seed LCQF: 12 tables in a 4x3 grid
-- Restaurant ID for LCQF: use subquery
DO $$
DECLARE
  lcqf_id uuid;
BEGIN
  SELECT id INTO lcqf_id FROM restaurants WHERE name ILIKE '%cabane%' LIMIT 1;
  IF lcqf_id IS NULL THEN
    RAISE NOTICE 'LCQF restaurant not found, skipping seed';
    RETURN;
  END IF;

  INSERT INTO restaurant_tables (restaurant_id, name, zone, capacity, shape, width, height, pos_x, pos_y, sort_order)
  VALUES
    (lcqf_id, 'T1',  'Salle', 4, 'square', 1, 1,  10, 10,  1),
    (lcqf_id, 'T2',  'Salle', 4, 'square', 1, 1,  30, 10,  2),
    (lcqf_id, 'T3',  'Salle', 4, 'square', 1, 1,  50, 10,  3),
    (lcqf_id, 'T4',  'Salle', 4, 'square', 1, 1,  70, 10,  4),

// ---------------------------------------------------------------------------
// Restaurant Tables

export interface RestaurantTable {
  id: string;
  restaurant_id: string;
  name: string;
  zone: string | null;
  capacity: number;
  shape: string;
  width: number;
  height: number;
  pos_x: number;
  pos_y: number;
  is_active: boolean;
  sort_order: number;
}

export async function getRestaurantTables(): Promise<RestaurantTable[]> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  const { data, error } = await supabase
    .from("restaurant_tables")
    .select("*")
    .eq("restaurant_id", restaurantId)
    .eq("is_active", true)
    .order("sort_order", { ascending: true });

  if (error) throw new Error(`Erreur chargement tables: ${error.message}`);
  return (data ?? []).map((t) => ({
    id: t.id,
    restaurant_id: t.restaurant_id,
    name: t.name,
    zone: t.zone,
    capacity: t.capacity ?? 4,
    shape: t.shape ?? "square",
    width: t.width ?? 1,
    height: t.height ?? 1,
    pos_x: t.pos_x ?? 0,
    pos_y: t.pos_y ?? 0,
    is_active: t.is_active ?? true,
    sort_order: t.sort_order ?? 0,
  }));
}

export async function upsertRestaurantTables(
  tables: {
    id?: string;
    name: string;
    zone?: string | null;
    capacity?: number;
    shape?: string;
    width?: number;
    height?: number;
    pos_x: number;
    pos_y: number;
    sort_order?: number;
  }[]
): Promise<void> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  for (const table of tables) {
    if (table.id) {
      // Update existing
      const { error } = await supabase
        .from("restaurant_tables")
        .update({
          name: table.name,
          zone: table.zone ?? null,
          capacity: table.capacity ?? 4,
          shape: table.shape ?? "square",
          width: table.width ?? 1,
          height: table.height ?? 1,
          pos_x: table.pos_x,
          pos_y: table.pos_y,
          sort_order: table.sort_order ?? 0,
        })
        .eq("id", table.id)
        .eq("restaurant_id", restaurantId);

      if (error) throw new Error(`Erreur update table ${table.name}: ${error.message}`);
    } else {
      // Insert new
      const { error } = await supabase
        .from("restaurant_tables")
        .insert({
          restaurant_id: restaurantId,
          name: table.name,
          zone: table.zone ?? null,
          capacity: table.capacity ?? 4,
          shape: table.shape ?? "square",
          width: table.width ?? 1,
          height: table.height ?? 1,
          pos_x: table.pos_x,
          pos_y: table.pos_y,
          sort_order: table.sort_order ?? 0,
        });

      if (error) throw new Error(`Erreur creation table ${table.name}: ${error.message}`);
    }
  }
}

export async function deleteRestaurantTable(tableId: string): Promise<void> {
  const restaurantId = await getUserRestaurantId();
  const supabase = await createClient();

  // Soft delete
  const { error } = await supabase
    .from("restaurant_tables")
    .update({ is_active: false })
    .eq("id", tableId)
    .eq("restaurant_id", restaurantId);

  if (error) throw new Error(`Erreur suppression table: ${error.message}`);
}

"use client";

import { cn } from "@/lib/utils";
import { Users } from "lucide-react";

type TableStatus = "free" | "occupied" | "waiting" | "ready" | "reserved";

interface TableShapeProps {
  name: string;
  shape: string;
  width: number;
  height: number;
  capacity: number;
  status?: TableStatus;
  isSelected?: boolean;
  onClick?: () => void;
  className?: string;
  /** Base size in pixels per grid unit (default 60) */
  baseSize?: number;
}

const STATUS_COLORS: Record<TableStatus, { bg: string; border: string }> = {
  free: { bg: "bg-green-100", border: "border-green-300" },
  reserved: { bg: "bg-purple-100", border: "border-purple-300" },
  occupied: { bg: "bg-blue-100", border: "border-blue-300" },
  waiting: { bg: "bg-amber-100", border: "border-amber-300" },
  ready: { bg: "bg-orange-100", border: "border-orange-300" },
};

export function TableShape({
  name,
  shape,
  width,
  height,
  capacity,
  status = "free",
  isSelected = false,
  onClick,
  className,
  baseSize = 60,
}: TableShapeProps) {
  const colors = STATUS_COLORS[status];
  const w = width * baseSize;
  const h = shape === "rectangle" ? height * baseSize : w;

  const borderRadius = shape === "round" ? "50%" : "12px";

  return (
    <button
      type="button"
      onClick={onClick}
      className={cn(
        "flex flex-col items-center justify-center border-2 transition-all hover:shadow-md active:scale-95",
        colors.bg,
        colors.border,
        isSelected && "ring-2 ring-primary ring-offset-2",
        status === "ready" && "animate-pulse",
        onClick && "cursor-pointer",
        className
      )}
      style={{
        width: `${w}px`,
        height: `${h}px`,
        borderRadius,
        minWidth: `${baseSize}px`,
        minHeight: `${baseSize}px`,
      }}
      aria-label={`Table ${name}`}
    >
      <span className="text-sm font-bold text-foreground leading-tight">{name}</span>
      <span className="flex items-center gap-0.5 text-[10px] text-muted-foreground">
        <Users className="h-2.5 w-2.5" />
        {capacity}
      </span>
    </button>
  );
}

export type { TableShapeProps, TableStatus };

      onDragEnd={handleDragEnd}

      <div className="flex h-full w-full gap-4 overflow-x-auto p-4">
        {COLUMNS.map((col) => {
          const columnTickets = grouped[col.key];
          return (
            <div
              key={col.key}
              className="flex min-w-[340px] flex-1 flex-col overflow-hidden rounded-xl bg-muted/30 ring-1 ring-foreground/5"
            >
              <div
                className={cn(
                  "flex items-center justify-between px-4 py-3",
                  col.headerBg,
                  col.headerText
                )}
              >
                <h2 className="text-lg font-bold">{col.title}</h2>
                <Badge
                  className={cn(
                    "min-w-[28px] justify-center text-sm font-bold",
                    col.badgeBg,
                    col.headerText,
                    "border border-white/30"
                  )}
                >
                  {columnTickets.length}
                </Badge>
              </div>

              <ScrollArea className="flex-1">
                <SortableContext
                  items={columnTickets.map((t) => t.id)}
                  strategy={verticalListSortingStrategy}
                >
                  <div className="space-y-3 p-3 min-h-[100px]">
                    {columnTickets.length === 0 ? (
                      <p className="py-12 text-center text-sm text-muted-foreground">
                        Aucune commande
                      </p>
                    ) : (
                      columnTickets.map((ticket) => (
                        <SortableTicket
                          key={ticket.id}
                          ticket={ticket}
                          isPulsing={pulsingIds.has(ticket.id)}
                          onItemStatusChange={onItemStatusChange}
                          onTicketStatusChange={onTicketStatusChange}
                          showStationBadge={showStationBadge}
                        />
                      ))
                    )}
                  </div>
                </SortableContext>
              </ScrollArea>
            </div>
          );
        })}
      </div>
    </DndContext>
  );
}
